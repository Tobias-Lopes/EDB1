int ListaEncadeada::removerTodos(int val) 
{
    No<int>* atual = getInicio();
    No<int>* anterior = nullptr;
    int qtd = 0;

    while (atual) {
        No<int>* proximo = atual->getProximo(); // salva o próximo antes de mexer

        if (atual->getValor() == val) {
            // Caso especial: removendo o primeiro nó
            if (anterior == nullptr) {
                setInicio(proximo);
            } else {
                anterior->setProximo(proximo);
            }

            delete atual;
            qtd++;
        } else {
            anterior = atual; // só atualiza anterior se não removeu
        }

        atual = proximo; // avança para o próximo nó sempre
    }

    return qtd;

}
